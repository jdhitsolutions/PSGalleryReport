    updatedITHosts = ITHosts | % { _.imagename = ''IT-image''}

    updateITHosts | % {
        jsonData = _ | ConvertTo-Json;
        Update-FogObject -Type object -CoreObject host -objectID _.id -jsonData jsonData;
