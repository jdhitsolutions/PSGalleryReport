```PowerShell
$right = [PSCustomObject]@{
    "lastname" = "von Bracht"

        "Street" = "Schaumainkai 87"

    }
    "tags" = [Array]@("wow")
    "product" = [PSCustomObject]@{
        "sprint" = 106
    }
}

```
