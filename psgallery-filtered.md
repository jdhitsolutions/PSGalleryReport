        "name" = "Orbit"
        "owner" = "Apteco Ltd."
    }

$right = [hashtable]@{
    "lastname" = "von Bracht"

    "address" = [hashtable]@{
        "Street" = "Schaumainkai 87"
        "Postcode" = 60596

    "tags" = [Array]@("wow")
    "product" = [PSCustomObject]@{
        "sprint" = 106
    }
}
